# 循环队列

循环队列

  1. 循环队列
  2. 根据击鼓传花的游戏进行计算

在击鼓传花的游戏中，孩子们围成一个圈，把花尽快传给旁边的人，

在某一时刻传花停止，这个时候话在谁手里谁淘汰。重复这个过程。

直到剩下一个孩子。

// 队列的构造函数
function Queue() {
  let items = [];

  // 向队列尾部添加一个或者多个新的项
  this.enqueue = function () {
    let args = arguments;

    for (let i = 0; i < args.length; i++) {
      items.push(args[i]);
    }
    return args[args.length - 1];
  };

  // 移除队列第一项，并返回此项的值
  this.dequeue = function () {
    return items.shift();
  };

  // 返回队列的第一个元素，不影响队列数据，只是返回值
  this.front = function () {
    return items[0];
  };

  // 如果是个空队列返回true，否则返回false
  this.isEmpty = function () {
    return items.length === 0;
  };

  // 清除队列中的元素
  this.clear = function () {
    items = [];
  };

  // 返回队列的元素个数
  this.size = function () {
    return items.length;
  };

  // 输出
  this.print = function () {
    return items.toString();
  };
}

//循环队列算法
function HotPotato(nameList, num) {
  let queue = new Queue();

  for (let i = 0; i < nameList.length; i++) {
    item = nameList[i];
    queue.enqueue(item);
  }

  let eliminated = "";
  while (queue.size() > 1) {
    for (let i = 0; i < num; i++) {
      queue.enqueue(queue.dequeue());
    }
    eliminated = queue.dequeue();
    console.log(eliminated + "在击鼓传花的游戏中淘汰");
  }

  return queue.dequeue();
}

let arr = ["A01", "A02", "A03", "A04", "A05"];
let winner = HotPotato(arr, 7);
console.log(winner + "是最终的胜利者");
