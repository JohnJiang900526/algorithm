# 创建队列

需求

  1. 创建一个队列
  2. 实现enqueue: 向队列尾部添加一个或者多个新的项
  3. 实现dequeue: 移除队列第一项，并返回此项的值
  4. 实现front: 返回队列的第一个元素，不影响队列数据，只是返回值
  5. 实现isEmpty: 如果是个空队列返回true，否则返回false
  6. 实现clear: 清除队列中的元素
  7. 实现size: 返回队列的元素个数

队列类或者构造函数

// 队列构造函数
function Queue() {
  let items = [];

  // 向队列尾部添加一个或者多个新的项
  this.enqueue = function () {
    let args = arguments;

    for (let i = 0; i < args.length; i++) {
      items.push(args[i]);
    }
    return args[args.length - 1];
  };

  // 移除队列第一项，并返回此项的值
  this.dequeue = function () {
    return items.shift();
  };

  // 返回队列的第一个元素，不影响队列数据，只是返回值
  this.front = function () {
    return items[0];
  };

  // 如果是个空队列返回true，否则返回false
  this.isEmpty = function () {
    return items.length === 0;
  };

  // 清除队列中的元素
  this.clear = function () {
    items = [];
  };

  // 返回队列的元素个数
  this.size = function () {
    return items.length;
  };

  // 输出
  this.print = function () {
    return items.toString();
  };
}


let queue = new Queue();

queue.enqueue(1, 2, 3, 4);
queue.dequeue();
queue.front();
queue.isEmpty();
queue.clear();
queue.size();
