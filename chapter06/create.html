# 创建集合

需求分析

  1. add(value): 向集合添加新的项
  2. remove(value): 在集合中移除一个值
  3. has(value): 如果值在集合中，返回true 否则false
  4. clear(): 清除集合的所有项
  5. size(): 返回集合中，元素的数量。
  6. values(): 返回有所值，数组形式展现。
// Set构造函数
function Set() {
  let items = {};

  // has(value): 如果值在集合中，返回true 否则false
  this.has = function (value) {
    return items.hasOwnProperty(value);
  };

  // add(value): 向集合添加新的项
  this.add = function (value) {
    if (this.has(value)) {
      return false;
    }

    items[value] = value;
    return true;
  };

  // remove(value): 在集合中移除一个值
  this.remove = function (value) {
    if (!this.has(value)) {
      return false;
    }

    delete items[value];
    return true;
  };

  // clear(): 清除集合的所有项
  this.clear = function () {
    items = {};
  };

  // size(): 返回集合中，元素的数量
  this.size = function () {
    let count = 0;
    for (let key in items) {
      count++;
    }
    return count;
  };

  // values(): 返回有所值，数组形式展现
  this.values = function () {
    let keys = [];

    for (let key in items) {
      keys.push(key);
    }
    return keys;
  };
}

// 应用
let set = new Set();
set.add(1);
set.add(2);
set.add(3);

console.log(set.has(1));
console.log(set.values());
console.log(set.size());

// true
// [ '1', '2', '3' ]
// 3
