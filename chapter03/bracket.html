# 平衡圆括号

注意

核心思想： 如果是"("就入栈，如果")"就弹出栈顶元素，最后看栈的长度是否为0；

function Stack() {
  let items = [];

  // 实现可以添加一个或者多个新元素到栈顶
  this.push = function () {
    let args = arguments;

    for (let i = 0; i < args.length; i++) {
      items[items.length] = args[i];
    }

    return args[args.length];
  };

  // 实现pop方法，移除栈顶元素。同时返回被移除的元素。
  this.pop = function () {
    return items.pop();
  };

  // 实现peek方法，返回栈顶元素，不会移除栈顶元素。
  this.peek = function () {
    return items[items.length - 1];
  };

  // 实现isEmpty方法，如果没有任何元素返回true，否则false
  this.isEmpty = function () {
    return items.length === 0;
  };

  // 实现clear方法，移除栈里的所有元素。
  this.clear = function () {
    items = [];
  };

  // 实现size方法，和数组的length属性类似。
  this.size = function () {
    return items.length;
  };

  // 实现print方法，输出整个队列信息。
  this.print = function () {
    return items.toString();
  };
}


// 圆括号平衡
function bracketBalance(str) {
  let stack = new Stack();
  var temp = str.split("");

  temp.forEach(function (item) {
    if (item == '(') {
      stack.push('(');
    } else if (item === ")") {
      if (stack.size() == 0) {
        return false;
      } else {
        stack.pop();
      }
    }
  });

  return stack.size() == 0;
}

console.log( bracketBalance("(())") ); // true
console.log( bracketBalance("((00") ); // false

// 多种括号平衡
function bracketBalance(str) {
  let stack = new Stack();
  let tempArr = str.split("");

  let leftMap = ["(", "{", "["];
  let rightMap = [")", "}", "]"];

  for (let i = 0; i < tempArr.length; i++) {
    let item = tempArr[i];
    if (leftMap.includes(item)) {
      stack.push(item);
    } else if (rightMap.includes(item)) {
      if (stack.size() === 0) {
        return false;
      }

      let popValue = stack.pop();
      switch(item) {
        case ")":
          if (popValue !== "(") {
            return false;
          }
          break;
        case "}":
          if (popValue !== "{") {
            return false;
          }
          break;
        case "]":
          if (popValue !== "[") {
            return false;
          }
          break;
      }
    }
  }

  return stack.size() === 0;
}

console.log(bracketBalance("([])")); // true
console.log(bracketBalance("()[")); // false
