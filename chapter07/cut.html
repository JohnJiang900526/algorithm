# 分离链接

分离链接

分离链接法包括为散列表的没一个位置创建一个链表，并将元素存储在链表中。

它是解决冲突的最简单的方法，但是它在HashTable 实例之外还需要额外的空间。

回顾链表

// 链表类
function LinkedList() {
  function Node(element) {
    this.element = element;
    this.next = null;
  };

  let length = 0;
  let head = null;

  // 实现append(element): 向列表尾部添加一个新的项
  this.append = function () {
    function addElement(element) {
      let node = new Node(element);
      let current;
  
      if (head === null) {
        head = node;
      } else {
        current = head;
  
        while(current.next) {
          current = current.next;
        }
  
        current.next = node;
      }
  
      length++;
    }

    let args = arguments;
    let lastElement;
    for (let i = 0; i < args.length; i++) {
      lastElement = args[i];
      addElement(lastElement);
    }
  
    return lastElement;
  };

  // 实现removeAt(position): 从特定的位置移除一项
  this.removeAt = function (position) {
    if (position > -1 && position < length) {
      let current = head;
      let previous;
      let index = 0;

      if (position === 0) {
        head = current.next;
      } else {
        while(index++ < position) {
          previous = current;
          current = current.next;
        }
        // 最关键的一步
        // 将previous的下一步和current的下一步连接起来
        // 从而删除了current
        previous.next = current.next;
      }

      length--;
      return current.element;
    } else {
      return null;
    }
  };

  // 实现insert(position, element): 向列表特定的位置插入一个新的项
  this.insert = function (position, element) {
    if (position >= 0 && position <= length) {
      let node = new Node(element);
      let current = head;
      let previous, index = 0;

      if (position === 0) {
        node.next = current;
        head = node;
      } else {
        // 核心代码
        // 查找指定位置的元素
        while (index++ < position) {
          previous = current;
          current = current.next;
        }
        // 位置置换
        node.next = current;
        previous.next = node;
      }

      length++;
      return true;
    } else {
      // 超出越界值的返回false
      return false;
    }
  };

  // 实现remove(element): 从列表中移除一项
  this.remove = function (element) {
    let index = this.indexOf(element);

    return this.removeAt(index);
  };

  // 实现indexOf(element): 返回元素在列表中的索引。如果不存在返回-1
  this.indexOf = function(element) {
    let current = head;
    let index = 0;

    while(current) {
      if (element === current.element) {
        return index;
      }

      index++;
      current = current.next;
    }
    return -1;
  };

  // 实现isEmpty(): 如果链表中不存在任何元素返回true，否则返回false
  this.isEmpty = function () {
    return length === 0;
  };

  // 实现size(): 返回链表元素个数，和length类似
  this.size = function() {
    return length;
  };

  // getHead
  this.getHead = function() {
    return head;
  };

  // 实现toString(): 输出对应元素的值
  this.toString = function () {
    let current = head;
    let string = "";

    while(current) {
      if (!string) {
        string = current.element;
      } else {
        string += ("," + current.element);
      }
      current = current.next;
    }

    return string;
  };
}

优化之后的HashTable

function HashTable() {
  let table = [];

  function loseloseHashCode(key) {
    let hash = 0;
    for (let i = 0; i < key.length; i++) {
      hash += key.charCodeAt(i);
    }
    return hash % 37;
  }

  function ValuePair(key, value) {
    this.key = key;
    this.value = value;

    this.toString = function () {
      return "["+ this.key + " - " + this.value +"]";
    };
  }

  // 实现put方法
  this.put = function (key, value) {
    let position = loseloseHashCode(key);

    if (table[position] == undefined) {
      table[position] = new LinkedList();
    }

    table[position].append(new ValuePair(key, value));
  };

  // 实现remove方法
  this.remove = function (key) {
    let position = loseloseHashCode(key);
    if (!table[position]) {
      return false;
    }

    let current = table[position].getHead();
    while (current.next) {
      if (current.element.key === key) {
        table[position].remove(current.element);
        if (table[position].isEmpty()) {
          table[position] = undefined;
        }
        return true;
      }
      current = current.next;
    }

    if (current.element.key === key) {
      table[position].remove(current.element);
      if (table[position].isEmpty()) {
        table[position] = undefined;
      }
      return true;
    }
    return false;
  };

  // 实现get方法
  this.get = function (key) {
    let position = loseloseHashCode(key);

    if (table[position] === undefined) {
      return undefined;
    }

    let current = table[position].getHead();
    while (current.next) {
      if (current.element.key === key) {
        return current.element.value;
      }
      current = current.next;
    }

    if (current.element.key === key) {
      return current.element.value;
    }
    return undefined;
  };

  // print
  this.print = function () {
    for (let i = 0; i < table.length; i++) {
      if (table[i] !== undefined) {
        console.log(table[i].toString());
      }
    }
  };
}
