# 字典

字典

集合表示一组互不相同的元素(不重复的元素)。

在字典中，存储的是键值对，其中键名是用来特定查询的元素。

字典和集合很相似，集合以值值的形式进行存储元素。

字典是以键值对来存储元素。字典也称为映射。

# 创建一个字典

字典需求

  1. set(key, value): 向字典中添加新的元素
  2. remove(key): 通过键删除字典中对应的元素
  3. has(key): 如果字典中存在此元素，返回true， 否则返回false
  4. get(key): 返回字典中键为key的值
  5. clear(): 将字典中的全部元素删除
  6. size(): 返回字典中的元素数量
  7. keys(): 返回字典中所有的key的一个数组
  8. values(): 返回字典中的所有元素值的一个数组

字典方法实现

function Dictionary() {
  let items = {};

  // set方法
  this.set = function (key, value) {
    items[key] = value;
  };

  // remove方法
  this.remove = function (key) {
    if (this.has(key)) {
      delete items[key];
      return true;
    }
    return false;
  };

  // has方法
  this.has = function (key) {
    return key in items;
  };

  // get方法
  this.get = function (key) {
    return items[key] ? items[key] : undefined;
  };

  // clear方法
  this.clear = function () {
    items = {};
  };

  // size方法
  this.size = function () {
    let count = 0;
    for (let key in items) {
      count++;
    }
    return count;
  };

  // keys 方法
  this.keys = function () {
    let arr = [];
    for (let key in items) {
      arr.push(key);
    }
    return arr;
  };

  // values方法
  this.values = function () {
    var arr = [];
    for (let key in items) {
      arr.push(items[key]);
    }
    return arr;
  };

  // getItems 方法
  this.getItems = function () {
    return items;
  };
}

字典调用

let dictionary = new Dictionary();

dictionary.set("key01", 1);
dictionary.set("key02", 2);
dictionary.set("key03", 3);
dictionary.set("key04", 4);
dictionary.set("key05", 5);

dictionary.remove("key02");
console.log(dictionary.keys());
console.log(dictionary.values());
console.log(dictionary.getItems());
console.log(dictionary.size());
console.log(dictionary.get("key03"));
// [ 'key01', 'key03', 'key04', 'key05' ]
// [ 1, 3, 4, 5 ]
// { key01: 1, key03: 3, key04: 4, key05: 5 }
// 4
// 3
