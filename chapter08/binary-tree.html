# 二叉树和二叉搜索树

二叉树和二叉搜索树

二叉树的节点 最多只能有两个子节点：一个是左侧子节点，一个是右侧子节点。

二叉搜索树(BST) 是二叉树的一种，它只允许左侧节点存储(比父节点)小的值， 右侧节点存储(比父节点大)或者(等于)的值。

二叉搜索树使我们本章节研究的重点。

需求分析

  1. insert(key): 向树中添加一个新的键。
  2. search(key): 在树中搜索一个键，如果存在返回true，否则返回false
  3. inOrderTraverse: 通过中序遍历所有节点。
  4. preOrderTraverse: 通过先序遍历所有节点。
  5. postOrderTraverse: 通过h后序遍历所有节点。
  6. min: 返回树中最小的键值。
  7. max：返回树中最大的键值。
  8. remove(key): 从树中移除某一个键。
// 二叉搜索树类(构造函数)的实现
function BinarySearchTree() {
  function Node(key) {
    this.key = key;
    this.left = null;
    this.right = null;
  }

  let root = null;

  // insert(key): 向树中添加一个新的键。
  this.insert = function (key) {
    let newNode = new Node(key);

    // 递归插入
    function insertNode(node, newNode) {
      if (newNode.key < node.key) {
        if (node.left === null) {
          node.left = newNode;
        } else {
          insertNode(node.left, newNode);
        }
      } else {
        if (node.right === null) {
          node.right = newNode;
        } else {
          insertNode(node.right, newNode);
        }
      }
    }

    if (root === null) {
      root = newNode;
    } else {
      insertNode(root, newNode);
    }
  };

  // search(key): 在树中搜索一个键，如果存在返回true，否则返回false
  this.search = function (key) {
    function searchNode(node, key) {
      if (node === null) {
        return false;
      }

      if (key < node.key) {
        return searchNode(node.left, key);
      } else if (key > node.key) {
        return searchNode(node.right, key);
      } else {
        return true;
      }
    }
    return searchNode(root, key);
  };

  // inOrderTraverse: 通过中序遍历所有节点。
  this.inOrderTraverse = function (fn) {
    function inOrderTraverseNode(node, fn) {
      if (node !== null) {
        inOrderTraverseNode(node.left, fn);
        fn && fn(node.key);
        inOrderTraverseNode(node.right, fn);
      }
    }

    inOrderTraverseNode(root, fn);
  };

  // preOrderTraverse: 通过先序遍历所有节点。
  this.preOrderTraverse = function (fn) {
    function preOrderTraverseNode(node, fn) {
      if (node !== null) {
        fn && fn(node.key);
        preOrderTraverseNode(node.left, fn);
        preOrderTraverseNode(node.right, fn);
      }
    }

    preOrderTraverseNode(root, fn);
  };

  // postOrderTraverse: 通过h后序遍历所有节点。
  this.postOrderTraverse = function (fn) {
    function postOrderTraverseNode(node, fn) {
      if (node !== null) {
        postOrderTraverseNode(node.left, fn);
        postOrderTraverseNode(node.right, fn);
        fn && fn(node.key);
      }
    }

    postOrderTraverseNode(root, fn);
  };

  // min: 返回树中最小的键值。
  this.min = function () {
    function minNode(node) {
      if (node) {
        while(node && node.left !== null) {
          node = node.left;
        }
        return node.key;
      } else {
        return null;
      }
    }

    return minNode(root);
  };

  // minNode: 返回树中最小的键值的节点。
  this.minNode = function () {
    function minNode(node) {
      if (node) {
        while(node && node.left !== null) {
          node = node.left;
        }
        return node;
      } else {
        return null;
      }
    }

    return minNode(root);
  };

  // max：返回树中最大的键值。
  this.max = function () {
    function maxNode(node) {
      if (node) {
        while(node && node.right !== null) {
          node = node.right;
        }
        return node.key;
      } else {
        return null;
      }
    }

    return maxNode(root);
  };

  // remove(key): 从树中移除某一个键。
  this.remove = function (key) {
    let that = this;
    function removeNode(node, key) {
      if (node === null) {
        return null;
      }

      if (key < node.key) {
        // 如果值小于当前节点值，递归其左子树
        node.left = removeNode(node.left, key);
        return node;
      } else if (key > node.key) {
        // 如果值大于当前节点值，递归其右子树
        node.right = removeNode(node.right, key);
        return node;
      } else {
        // 第一种情况 如果值等于当前节点值，做该节点删除操作
        if (node.left === null && node.right === null) {
          node = null;
          return node;
        }

        // 第二种情况 只有一个子节点
        if (node.left === null) {
          // 如果该节点只有右子树，则将该节点指向它的右子树
          node = node.right;
          return node;
        } else if (node.right === null) {
          // 如果该节点只有左子树，则将该节点指向它的左子树
          node = node.left;
          return node;
        }

        // 第三种情况 一个具有两个子节点的节点
        // 如果该节点有左右子树，则把找到它的右子树的最小子节点，
        // 并使该节点的值等于它的右子树的最小子节点的值，然后删除它的右子树的最小子节点
        let aux = that.minNode(node.right);
        node.key = aux.key;
        node.right = removeNode(node.right, aux.key);
        return node;
      }
    }
    root = removeNode(root, key);
  };

  // 获取根节点
  this.getRootNode = function () {
    return root;
  };
}

// 实例代码
let tree = new BinarySearchTree();

tree.insert(11);
tree.insert(7);
tree.insert(15);
tree.insert(5);
tree.insert(3);
tree.insert(9);
tree.insert(8);
tree.insert(10);
tree.insert(13);
tree.insert(12);
tree.insert(14);
tree.insert(20);
tree.insert(18);
tree.insert(25);
tree.insert(6);

tree.remove(11);

tree.remove(11);

); // true
